Notes We first define expressed genes in the Macrophages or T cells and NK cells (sender) and malignant cells (receiver). Genes are considered to be expressed within one cell type if their log2-transformed mean TPM (transcript per kilobase million) values are greater than 4, following the same criterion as in the NicheNet tutorial. The mean TPM value for each cell type is calculated based on the cell-type-specific gene expression of spots with the corresponding proportion greater than 0.05 within the focused spots.

# Define genes and filter spots based on criteria
keep_genes <- rownames(stged$F_list[[1]])
keep_spots_Macro <- stged$beta[,5] > 0.05
keep_spots_Malignant <- stged$beta[,6] > 0.05
keep_spots_T <- stged$beta[,7] > 0.05

# Identify expressed genes for T cells, macrophages, and malignant cells
expressed_genes_T <- keep_genes[log2(rowMeans(stged$F_list[[7]][,keep_spots_T, drop = FALSE] * 1e6) + 1) >= 4]
expressed_genes_Macro <- keep_genes[log2(rowMeans(stged$F_list[[5]][,keep_spots_Macro, drop = FALSE] * 1e6) + 1) >= 4]
expressed_genes_Malignant <- keep_genes[log2(rowMeans(stged$F_list[[6]][,keep_spots_Malignant, drop = FALSE] * 1e6) + 1) >= 4]
