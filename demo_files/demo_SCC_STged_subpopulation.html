generate_plot <- function(cluster_id, markers_cluster, spot_loc, manigant_cluster_sec, spot_lable) {
  exp = GetAssayData(object = manigant_cluster_sec, assay = "RNA", layer = "counts")[markers_cluster$gene, 
                                                                                   WhichCells(manigant_cluster_sec, idents = cluster_id)]
  
  exp_mean = apply(exp, 2, mean)
  spot_lable_sec = spot_lable[names(exp_mean)]
  
  meta_plot = spot_loc
  meta_plot$prop = 0
  meta_plot[spot_lable_sec,]$prop = exp_mean
  
  plots_sub = ggplot(data = meta_plot, 
                     mapping = aes(x = x, y = y, color = prop)) +
    geom_point(size = 1) + 
    scale_color_gradientn(colours = c("darkblue", "blue", "white", "red", "darkred")) +
    theme_bw() + 
    labs(y = "", x = " ", title = cluster_id, color = "", size = "") +
    theme(
      axis.ticks = element_blank(), 
      axis.text = element_blank(),
      panel.grid.major = element_blank(),
      panel.grid.minor = element_blank(),
      panel.border = element_blank(),
      axis.line = element_blank(),
      plot.title = element_text(hjust = 0.5, size = 12) ,
      panel.background = element_blank()
    )
  
  return(plots_sub)
}

# Step 2: Perform Permutation Analysis
n_permutations <- 1000
perm_results <- list()
set.seed(42)

for (i in 1:n_permutations) {
  permuted_meta_plot <- meta_plot
  permuted_meta_plot$region <- sample(meta_plot$region)
  
  perm_enrichment_table <- table(permuted_meta_plot$region, permuted_meta_plot$prop)
  perm_enrichment_df <- as.data.frame(perm_enrichment_table)
  colnames(perm_enrichment_df) <- c("region", "cluster", "Freq")
  
  perm_enrichment_df$prop_region <- with(perm_enrichment_df, Freq / ave(Freq, cluster, FUN = sum))
  perm_results[[i]] <- perm_enrichment_df
}

perm_df <- do.call(rbind, perm_results)

# Step 3: Calculate Enrichment Direction and Effect Size
enrichment_summary <- data.frame(region = character(), cluster = character(), 
                                 p_value = numeric(), effect_size = numeric(), 
                                 Direction = character(), stringsAsFactors = FALSE)

for (region in unique(enrichment_df$region)) {
  for (cluster in unique(enrichment_df$cluster)) {
    # Observed value for this region-cluster pair
    observed_value <- enrichment_df$prop_region[enrichment_df$region == region & enrichment_df$cluster == cluster]
    
    # Permuted values for this region-cluster pair
    perm_values <- perm_df$prop_region[perm_df$region == region & perm_df$cluster == cluster]
    
    # Calculate p-value and determine enrichment direction
    p_value <- mean(perm_values >= observed_value)
    direction <- ifelse(p_value < 0.05, "Enriched", "Depleted")
    
    enrichment_summary <- rbind(enrichment_summary, 
                                data.frame(region = region, cluster = cluster, 
                                           p_value = p_value, effect_size = observed_value, 
                                           Direction = direction))
  }
}
